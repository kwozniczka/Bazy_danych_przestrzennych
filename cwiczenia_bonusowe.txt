SELECT mecze.* FROM mecze 
JOIN gole ON mecze.id = gole.matchid 
WHERE id = '1004' 
AND player = 'Jakub Blaszczykowski';

SELECT player, teamname, stadium, mdate FROM mecze
JOIN gole ON id = matchid 
JOIN druzyny ON druzyny.id = teamid 
WHERE teamname = 'Poland'
